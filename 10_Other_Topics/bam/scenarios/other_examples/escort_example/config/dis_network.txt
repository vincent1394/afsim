###############################################################################
#                            DIS Network definitions
#
# This file can be 'include'd into simulations that want to participate in a
# DIS exercise. It defines the default DIS network parameters that are to be
# used by the application.
#
# If you want to run multiple instances of SAGE in an exercise, they each
# instance would probably 'include' this file and then follow with another
# 'dis_interface' block to override the 'application'.
#
# For example, the input to first instance would specify:
#
#     include dis_network.txt
#     dis_interface
#       application 2
#     end_dis_interface
#
# and the input to the second instance would specify:
#
#     include dis_network.txt
#     dis_interface
#       application 3
#     end_dis_interface
###############################################################################

#include_once entity_map.txt
#include_once force_map.txt
#include_once emitter_map.txt

dis_interface
  multicast   229.1.1.1   192 
  port                 3271

  # Uncomment the 'autostart' line if the clock is to start automatically.
  # If this line in not uncommented then the application will wait until it
  # receives a DIS Start PDU from a simulation controller.

  #############################################################################
  # All of the following have default values.  They are provided here just to
  # document the command formats.
  #############################################################################

  exercise                     1               # default is 1
  site                         1               # default is 1
  application                  1               # default is 1

  # Physical time is multplied by the following to generate simulation time.

  heartbeat_timer              5.0 seconds
  heartbeat_multiplier         2.4

  entity_position_threshold    1 m             # default is 1 m
  entity_orientation_threshold 3 deg           # default is 3 degrees

  protocol_version             6               # default is 1278.1
                                               # 1 = 1.0
                                               # 2 = IEEE 1278 (1993)
                                               # 3 = 2.0 Third Draft
                                               # 4 = 2.0 Fourth Draft
                                               # 5 = IEEE 1278.1 (1995)
                                               # 6 = IEEE 1278.1a (1998)
                                               
  //mover_update_timer 5 seconds
  
  #debug_emission_pdu 2
  
  autostart

end_dis_interface

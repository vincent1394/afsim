script_variables
   bool val_a = true; 
   bool val_b = false; 
   bool val_c = true; 
   bool val_d = false; 
   bool val_e = false; 
   bool val_f = true; 
   bool val_g = false; 
   bool val_h = true; 
end_script_variables

behavior a precondition return val_a; end_precondition execute writeln(" a ran!"); end_execute end_behavior 
behavior b precondition return val_b; end_precondition execute writeln(" b ran!"); end_execute end_behavior 
behavior c precondition return val_c; end_precondition execute writeln(" c ran!"); end_execute end_behavior 
behavior d precondition return val_d; end_precondition execute writeln(" d ran!"); end_execute end_behavior 
behavior e precondition return val_e; end_precondition execute writeln(" e ran!"); end_execute end_behavior 
behavior f precondition return val_f; end_precondition execute writeln(" f ran!"); end_execute end_behavior 
behavior g precondition return val_g; end_precondition execute writeln(" g ran!"); end_execute end_behavior 
behavior h precondition return val_h; end_precondition execute writeln(" h ran!"); end_execute end_behavior 

platform q1 WSF_PLATFORM

   add processor tree WSF_SCRIPT_PROCESSOR
      update_interval 50 sec
      on_update writeln(""); end_on_update
      behavior_tree
         parallel
            sequence
               selector
                  behavior_node a
                  behavior_node b
               end_selector
               behavior_node c
            end_sequence
            selector
               behavior_node d
               behavior_node e
            end_selector
            sequence
               behavior_node f
               selector
                  behavior_node g
                  behavior_node h
               end_selector
            end_sequence
         end_parallel
      end_behavior_tree
   end_processor
end_platform

event_output
   file output/bt_quiz_2.evt
   time_format h:m:s.1

   enable BTREE_NODE_CHILDREN
   enable BTREE_NODE_EXEC
end_event_output

end_time 100 sec
